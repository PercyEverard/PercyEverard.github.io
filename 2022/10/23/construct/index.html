c++ support the inherit through class derivation.The essays about inherit are put on before.This time i wanna show you an instance in regard to how the construct function works while inherit is happening...

/#include <iostream>
/#include <string>
using namespace std;
enum Gender{male,female,none};
struct course{
	string coursename;
	int grade;
};
class Person{
	//string Name;
	//Gender Sex;
	//int Age;
	//if writing here means private scope
public: 
	string Name;
	Gender Sex;
	int Age;
	Person(string, Gender, int);
	Person();
	~Person();
	void printf();
};
Person::Person(string name, Gender sex, int age)
{
	Name = name;
	Sex = sex;
	Age = age;
}
Person::Person()
{
	Name = "#";
	Sex = none;
	Age = 0;
}
Person::~Person(){}
void Person::printf()
{
	cout << "Name:" << Name << ", Gender:" << Sex << ",Age:" << Age << endl;
}
class Student:public Person
{
	//string name;
	course cs[3];
public:
	Student(string name, Gender sex, int age);
	Student();
	~Student();
	void printf_student();
};
//Attention! parameters name must be the same...
Student::Student(string name, Gender sex, int age)	
	:Person(name, sex, age)
{
	int i;
	for(i = 0;i < 3; i++)
	{
		cs[i].coursename = "#";
		cs[i].grade = 0;
	}
}
//Attention!The base class parameterless constructor doest need to explicitly given!
Student::Student()
{
	int i;
	for(i = 0;i < 3;i++)
	{
		cs[i].coursename = "#";
		cs[i].grade = 0;
	}
}
Student::~Student(){}
void Student::printf_student()
{
	cout << "Id:" << Name << ", Gender:" << Sex << ",Age:" << Age << endl;
	//cout << "Id:" << id << endl;
	cout << "---------------------------------------" << endl;
	int i ;
	for (i = 0; i < 3; i++)
	{
		cout << "coursename:" << cs[i].coursename << ",grade:" << cs[i].grade << endl;
	}
}
int main()
{
	Person wenfeng("wenfeng", male, 21);
	wenfeng.printf();
	Student wenhui("3120002382", male, 10);
	wenhui.printf_student();
	return 0;
}
//By the way, this is the explicit inheritance we call for.
